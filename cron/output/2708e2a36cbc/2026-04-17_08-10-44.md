```python
# ⚠️ 重要：腾讯接口返回 GBK 编码，不是 UTF-8！
# 请求示例
indices_url = "https://qt.gtimg.cn/q=sh000001,sz399001,sz399006,sh000300,sh000688"

# 解析: parts[3]=当前价, parts[32]=涨跌幅
# ⚠️ 注意：response格式是 v_sh000001="1~name~code~price~..." 需要先用 '="' 分割，再用 '~' 分割值
def parse_tencent_realtime(raw_gbk_text):
    result = {}
    for line in raw_gbk_text.strip().split('\n'):
        if '="' not in line:
            continue
        code_part = line.split('="')[0].replace('v_', '')  # e.g. "sh000001"
        value = line.split('="')[1].rstrip('"').rstrip(';').rstrip('"')  # 去掉末尾 ";]"
        parts = value.split('~')
        if len(parts) > 32:
            name = parts[1]
            price = float(parts[3]) if parts[3] else 0
            pct = float(parts[32]) if parts[32] else 0
            result[code_part] = {'name': name, 'price': price, 'pct': pct}
    return result
```

## 消息面时间规则（重要！）

### 判断逻辑
```python
from datetime import datetime, timedelta

def get_news_start_time():
    """计算消息面开始时间
    
    今天周四(4月16日) -> 消息从周三(4月15日) 08:10 开始
    今天周一       -> 消息从上周五 08:10 开始
    """
    today = datetime.now()
    weekday = today.weekday()  # 0=周一, 6=周日
    
    if weekday == 0:  # 周一
        # 周一：消息从上一个交易日的"前一天"早上8:10开始
        # 即：上周五 08:10
        news_start = today - timedelta(days=3)
    else:
        # 周二至周五：消息从前一个交易日早上8:10开始
        news_start = today - timedelta(days=1)
    
    news_start = news_start.replace(hour=8, minute=10, second=0)
    return news_start
```

## 定时任务配置

### ⚠️ 时区关键问题
**cron系统使用UTC时间，不是本地时间！**

用户期望的"北京时间早上7:30"换算成UTC：
- 北京时间 07:30 = UTC 23:30 (前一天)
- 所以 cron 表达式是 `30 23 * * 0-4`
  - `30 23` = 23:30 UTC
  - `0-4` = 周日(0)、周一(1)、周二(2)、周三(3)、周四(4)
  - 23:30 UTC周日 = 周一07:30北京时间
  - 23:30 UTC周一 = 周二07:30北京时间
  - ...以此类推
  - 周五07:30北京时间 = 周四23:30 UTC → 用 `30 23 * * 0-4` 覆盖
